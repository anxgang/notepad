# simple form

PDF 列印工具

# 安裝

gem 'prawn', '~> 2.2.2'
gem 'prawn-table', '~> 0.2.2'

# 官方手冊

  • prawn PDF
  • prawn-table PDF

# 簡易使用方式

respond_to do |format|
  format.html
  format.pdf do
    pdf = Prawn::Document.new do
      # 註冊字型(中文必須)
      font_families.update( 
        "NotoSans" => { 
          normal: Rails.root.join('app','assets','fonts', 'NotoSansTC-Medium.ttf')
        }
      )
      # 預設字型
      font "NotoSans"

      # 插入圖片
      image Rails.root.join('app','assets','images','kixpress.jpg'), :fit => [125, 125]

      # 直接輸出
      move_down 10
      text "訂單編號:"

      move_down 10
      text "物流設定:"

    end

    #輸出
    # disposition:inline 可以直接使用chrome開啟，不是下載
    send_data pdf.render, type: 'application/pdf', disposition: 'inline'
  end
end
